发布日期:   2022-08-04 
import axios from "axios";
import { RESPONSECODE } from "@/constant/index";
import { debounce, objValTrimFormat } from "./utils";
import router from "@/router/index.js";
import { useUserStore } from "@/stores/user";
import { ElMessage, ElMessageBox } from "element-plus";

const service = axios.create({
  timeout: 60000 * 2,
  headers: {
    platform: "web",
    appId: import.meta.env.VITE_APP_APPLICATION_ID
  }
});
const errBody = { code: -1, message: "网络异常" };

const requestList = [];

const { CancelToken } = axios;

service.interceptors.request.use(
  (config) => {
    const { cancelToken, notToken } = config;
    const { token, deviceId } = useUserStore().authInfo || {};
    // 某些接口带token报错
    if (!notToken && token) {
      config.headers.token = token;
      config.headers.deviceId = deviceId;
    }
    if (cancelToken === undefined) {
      // 禁止自动取消请求设置cancelToken=false
      // 自定义控制取消时机则正常设置cancelToken
      const { find, cancelFun, cancelKey } = cancelRequest(config);
      if (find) {
        cancelFun(`取消请求${cancelKey}`);
        return config;
      }
      config.cancelToken = new CancelToken((cancel) => {
        requestList.push({ cancel, key: cancelKey });
      });
    }
    return config;
  },
  (error) => {
    return Promise.reject(error);
  }
);
let debounceFunc;
service.interceptors.response.use(
  ({ data, config }) => {
    cancelRequest(config);
    return data;
  },
  (error) => {
    if (axios.isCancel(error)) {
      console.warn(error.message);
      return Promise.resolve({});
    } else {
      if (!debounceFunc) {
        debounceFunc = debounce(ElMessageBox.bind(this, { title: "系统提示", message: "网络异常，请稍后再试", type: "warning" }), 1000 * 5);
      }
      debounceFunc();
      cancelRequest(error.config);
      return Promise.resolve(errBody);
    }
  }
);
/**
* http
*/
const http = async ({ showMsg, sMsg, eMsg, ...config }) => {
  const res = await service(config);
  if (res) {
    formatResMsg(res, { showMsg, sMsg, eMsg });
  }
  return res;
};
/**
* 返回code
* @param {*} res
* @param {showMsg:undefined|false,sMsg:undefined|string|false,eMsg:undefined|string|false} msg
*/
let authOut = true;
const formatResMsg = ({ code, message }, { showMsg, sMsg, eMsg }) => {
  switch (true) {
    case RESPONSECODE.TOKENOUT.includes(code):
      if (authOut) {
        const { logout } = useUserStore();
        logout();
        authOut = false;
        if (router.currentRoute.value.name !== "Login") {
          ElMessageBox.confirm("登录失效，请重新登录")
            .then(() => {
              router.push({ name: "Login" });
            })
            .finally(() => {
              authOut = true;
            });
        }
        requestList.forEach((xhr) => xhr.cancel());
      }
      break;
    case code !== undefined && showMsg !== false: {
      const hideSucc = sMsg === false;
      const hideError = eMsg === false;
      switch (true) {
        case RESPONSECODE.SUCCESS === code && !hideSucc && !!(sMsg || message):
          ElMessage.closeAll("success");
          ElMessage.success(sMsg || message);
          break;
        case RESPONSECODE.SUCCESS !== code && !hideError && !!(eMsg || message):
          ElMessage.closeAll();
          ElMessage.warning(eMsg || message);
          break;
        default:
          break;
      }
      break;
    }
    default:
      break;
  }
};

const request = {
  /**
  * GET 请求
  * @param {string} url 请求路径
  * @param {any} params 请求参数
  * @param {{showMsg:undefined|false,sMsg:undefined|string|false,eMsg:undefined|string|false,[key:string]:any}} confg
  * showMsg     设置是否弹出提示【默认弹出提示】
  *
  *         false: 禁止弹出提示
  *
  * sMsg          自定义成功提示【默认提示接口返回msg】
  *
  *         string: 自定义接口成功提示
  *
  *         false: 禁止弹出成功提示
  *
  * eMsg         自定义成功提示【默认提示接口返回msg】
  *
  *         string: 自定义接口失败提示
  *
  *         false: 禁止弹出失败提示
  *
  * [key: string] 自定义覆盖或添加 axios config 参数
  * @returns
  */
  get(url, params, confg) {
    return http({ ...confg, url, method: "GET", params: objValTrimFormat(params) });
  },
  /**
  * POST 请求
  * @param {string} url 请求路径
  * @param {any} data 请求参数
  * @param {{showMsg:undefined|false,sMsg:undefined|string|false,eMsg:undefined|string|false,[key:string]:any}} confg 同 GET 请求
  * @returns
  */
  post(url, data, confg) {
    return http({ ...confg, url, method: "POST", data: objValTrimFormat(data) });
  },
  /**
  * PUT 请求
  * @param {string} url 请求路径
  * @param {any} data 请求参数
  * @param {{showMsg:undefined|false,sMsg:undefined|string|false,eMsg:undefined|string|false,[key:string]:any}} confg 同 GET 请求
  * @returns
  */
  put(url, data, confg) {
    return http({ ...confg, url, method: "PUT", data: objValTrimFormat(data) });
  },
  /**
  * PATCH 请求
  * @param {string} url 请求路径
  * @param {any} data 请求参数
  * @param {{showMsg:undefined|false,sMsg:undefined|string|false,eMsg:undefined|string|false,[key:string]:any}} confg 同 GET 请求
  * @returns
  */
  patch(url, data, confg) {
    return http({ ...confg, url, method: "PATCH", data: objValTrimFormat(data) });
  },
  /**
  * DELETE 请求
  * @param {string} url 请求路径
  * @param {any} data 请求参数
  * @param {{showMsg:undefined|false,sMsg:undefined|string|false,eMsg:undefined|string|false,[key:string]:any}} confg 同 GET 请求
  * @returns
  */
  delete(url, data, confg) {
    return http({ ...confg, url, method: "DELETE", data: objValTrimFormat(data) });
  }
};
export default request;

function cancelRequest(config) {
  const { method, url, data, params } = config;
  const key = [method, url, JSON.stringify(data), JSON.stringify(params)].join(":");
  const index = requestList.findIndex((request) => {
    return request.key === key;
  });
  if (index >= 0) {
    const [cancelFun] = requestList.splice(index, 1);
    return { find: true, cancelFun: cancelFun.cancel, cancelKey: key };
  }
  return { find: false, cancelKey: key };
}
